    <script src="https://stackpath.bootstrapcdn.com/bootstrap/4.1.3/js/bootstrap.min.js" integrity="sha384-ChfqqxuZUCnJSK3+MXmPNIyE6ZbWh2IMqE241rYiqJxyMiZ6OW/JmZQ5stwEULTy" crossorigin="anonymous"></script>

    
    <script>

        $(document).ready(function(){


            var btnSalvar = document.getElementById("btnSalvar");
            var btnConsultar = document.getElementById("btnConsultar");
            var form = document.getElementById("form");

            btnSalvar.addEventListener("click", function(e){
                e.preventDefault();

                var name = document.getElementById("name").value;
                var usuario = document.getElementById("usuario").value;

                var obj = {

                    name : name,
                    usuario : usuario

                }

                localStorage.setItem("Cadastro", JSON.stringify(obj));
                form.reset();

            });

            btnConsultar.addEventListener("click", function(e){
                e.preventDefault();

                var dados_str = localStorage.getItem("Cadastro");
                var dados_obj = JSON.parse(dados_str);

                document.getElementById("name").value = dados_obj.name;
                document.getElementById("usuario").value = dados_obj.usuario;

            });






        });
    </script>
